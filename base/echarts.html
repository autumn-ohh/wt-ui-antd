    data() {
        return {
            chartOptions: {
                name: '饼图',

                        type: 'pie',
                        radius: ['40%', '70%'],
                        avoidLabelOverlap: false,
                        label: {
                            show: false,
                            position: 'center'

                            label: {
                                show: true,

}

    <div>
        <wt-echarts :chartOptions="chartOptions" />
    </div>
</template>

export default {
    data() {
        return {
            chartOptions: {
                name: '饼图',
                chartData: [

                        type: 'graph',

                            normal: {
                                show: true,

                                formatter: '{c}'

                        data: [
                            {
                                name: '人员1',
                                id: '1-1',
                                category: 1,
                                draggable: true
